## 1. Basic parent-child hierarchy

Indenting a table under another automatically creates a foreign key from child to parent. The child table gets a `<parent>_id` column and an index on that column.

**Input:**

```quicksql
departments /insert 2
  name /nn
  location
  employees /insert 4
    name /nn vc50
    email /lower
    salary num(10,2)
    date hired

emp_v = departments employees
```

**Key generated DDL:**

```sql
create table departments (
    id          number generated by default on null as identity
                constraint departments_id_pk primary key,
    name        varchar2(255 char) not null,
    location    varchar2(4000 char)
);

create table employees (
    id               number generated by default on null as identity
                     constraint employees_id_pk primary key,
    department_id    number
                     constraint employees_department_id_fk
                     references departments,
    name             varchar2(50 char) not null,
    email            varchar2(255 char),
    salary           number(10,2),
    date_hired       date
);

create index employees_i1 on employees (department_id);

-- /lower generates a BEFORE INSERT OR UPDATE trigger
create or replace trigger employees_bi
    before insert on employees for each row
begin
    :new.email := lower(:new.email);
end employees_bi;
/

-- Join view via "name = t1 t2" syntax
create or replace view emp_v as
select departments.id        department_id,
       departments.name      department_name,
       departments.location  location,
       employees.id          employee_id,
       employees.name        employee_name,
       employees.email       email,
       employees.salary      salary,
       employees.date_hired  date_hired
from   departments
left join employees on employees.department_id = departments.id;
```

## 2. Audit columns and APEX integration

`auditcols: yes` adds four audit columns to every table. The `apex: yes` setting switches the trigger to use `coalesce(sys_context('APEX$SESSION','APP_USER'), user)` instead of just `user`.

**Input:**

```quicksql
projects
  name vc200 /nn
  status /check OPEN,ACTIVE,CLOSED /nn
  budget num(12,2)

# settings = { auditcols: yes, apex: yes }
```

**Key generated DDL:**

```sql
create table projects (
    id            number generated by default on null as identity
                  constraint projects_id_pk primary key,
    name          varchar2(200 char) not null,
    status        varchar2(4000 char) not null
                  constraint projects_status_ck check (status in ('OPEN','ACTIVE','CLOSED')),
    budget        number(12,2),
    created       date,
    created_by    varchar2(255 char),
    updated       date,
    updated_by    varchar2(255 char)
);

create or replace trigger projects_biu
    before insert or update on projects for each row
begin
    if inserting then
        :new.created    := sysdate;
        :new.created_by := coalesce(sys_context('APEX$SESSION','APP_USER'), user);
    end if;
    :new.updated    := sysdate;
    :new.updated_by := coalesce(sys_context('APEX$SESSION','APP_USER'), user);
end projects_biu;
/
```

## 3. Column prefix and composite primary key

`/colprefix` adds a short prefix to every column in a table, reducing name collisions when joining. A table-level `/pk` builds a composite primary key from the listed columns (here used alongside manual `/pk` directives on columns).

**Input:**

```quicksql
order_lines /colprefix ol
  order_id   /fk orders /nn
  product_id /fk products /nn
  quantity   num /nn
  unit_price num(10,2) /nn
  /pk order_id product_id
```

**Key generated DDL:**

```sql
create table order_lines (
    ol_order_id      number not null
                     constraint order_lines_ol_order_id_fk
                     references orders,
    ol_product_id    number not null
                     constraint order_lines_ol_product_id_fk
                     references products,
    ol_quantity      number not null,
    ol_unit_price    number(10,2) not null,
    constraint order_lines_pk primary key (ol_order_id, ol_product_id)
);
```

The underscore separator is added automatically if not included in the prefix value.

## 4. Foreign key to a table outside the model

`/fk <table>` creates a foreign key to any table ŌĆö it does not need to be defined in the current QSQL script. This is useful when referencing lookup tables maintained separately.

**Input:**

```quicksql
invoices
  customer_id /fk customers /nn
  currency_code vc3 /fk ref_currencies /nn
  total num(14,2) /nn
  status /check DRAFT,ISSUED,PAID,VOID /nn
```

**Key generated DDL:**

```sql
create table invoices (
    id               number generated by default on null as identity
                     constraint invoices_id_pk primary key,
    customer_id      number not null
                     constraint invoices_customer_id_fk
                     references customers,
    currency_code    varchar2(3 char) not null
                     constraint invoices_currency_code_fk
                     references ref_currencies,
    total            number(14,2) not null,
    status           varchar2(4000 char) not null
                     constraint invoices_status_ck
                     check (status in ('DRAFT','ISSUED','PAID','VOID'))
);
```

If the referenced table does not exist at parse time QuickSQL still generates the constraint; the FK will validate at install time.

## 5. Star schema (fact + dimension tables)

The `>` prefix in front of a child table name reverses the FK direction: instead of the child carrying a FK to the parent, the **fact** table carries FKs to the **dimension** tables. This is the standard star-schema pattern.

**Input:**

```quicksql
sales /insert 2
  quantity
  > products /insert 1
    name
    category vc100
  > customers /insert 1
    first name
    country vc2

# settings = { pk: identityDataType, semantics: CHAR }
```

**Key generated DDL:**

```sql
-- Dimension tables have their own PKs
create table customers (
    id           number generated by default on null as identity
                 constraint customers_id_pk primary key,
    first_name   varchar2(4000 char),
    country      varchar2(2 char)
);

create table products (
    id          number generated by default on null as identity
                constraint products_id_pk primary key,
    name        varchar2(255 char),
    category    varchar2(100 char)
);

-- Fact table carries the FKs to both dimensions
create table sales (
    id              number generated by default on null as identity
                    constraint sales_id_pk primary key,
    product_id      number
                    constraint sales_product_id_fk references products,
    customers_id    number
                    constraint sales_customers_id_fk references customers,
    quantity        number
);

create index sales_i1 on sales (product_id);
create index sales_i2 on sales (customers_id);
```

## 6. JSON Duality Views (23ai+)

`dv <view_name> <root_table> <child_table>` generates a JSON Relational Duality View. The root table is the top-level JSON document; nested tables are embedded as arrays based on their FK relationships.

**Input:**

```quicksql
departments
  name vc200 /nn
  employees
    first_name vc100 /nn
    last_name  vc100 /nn
    salary     num(10,2)

dv dept_emp_dv departments employees

# settings = { db: "23c" }
```

**Key generated DDL:**

```sql
create table departments (
    id      number generated by default on null as identity
            constraint departments_id_pk primary key,
    name    varchar2(200 char) not null
);

create table employees (
    id               number generated by default on null as identity
                     constraint employees_id_pk primary key,
    department_id    number
                     constraint employees_department_id_fk
                     references departments,
    first_name       varchar2(100 char) not null,
    last_name        varchar2(100 char) not null,
    salary           number(10,2)
);

create or replace json relational duality view dept_emp_dv as
departments @insert @update @delete {
    _id       : id,
    name      : name,
    employees : employees @insert @update @delete
    [{
        _id        : id,
        first_name : first_name,
        last_name  : last_name,
        salary     : salary
    }]
};
```

Each row in `departments` becomes a JSON document. The `employees` array is embedded and fully updateable via the duality view.

## 7. Multi-lingual columns with /trans

`/trans` on a column generates three objects: a shared `language` lookup table, a `<table>_trans` translation table (one row per parent row ├Ś language), and a `<table>_resolved` view that applies `COALESCE` to fall back to the base value when no translation exists.

**Input:**

```quicksql
products
  name       vc200 /nn /trans
  description       /trans
  price      num(10,2)

# prefix: shop
```

**Key generated DDL (abbreviated):**

```sql
create table shop_products (
    id           number generated by default on null as identity
                 constraint shop_products_id_pk primary key,
    name         varchar2(200 char) not null,   -- base/fallback value
    description  varchar2(4000 char),           -- base/fallback value
    price        number(10,2)
);

-- Shared language table (one per schema)
create table shop_language (
    code           varchar2(5 char) not null
                   constraint shop_language_code_pk primary key,
    locale         varchar2(28 char) not null
                   constraint shop_language_locale_unq unique,
    name           varchar2(1024 char),
    native_name    varchar2(1024 char)
);

-- Translation table: trans_ prefix on translated columns
create table shop_products_trans (
    id               number generated by default on null as identity
                     constraint shop_products_trans_id_pk primary key,
    products_id      number not null,
    language_code    varchar2(5 char) not null,
    trans_name       varchar2(200 char),
    trans_description varchar2(4000 char),
    constraint shop_products_trans_uk unique (products_id, language_code)
);

-- Resolved view: LEFT JOIN with COALESCE fallback
create or replace view shop_products_resolved as
select k.id,
       coalesce(t.trans_name,        k.name)        as name,
       coalesce(t.trans_description, k.description) as description,
       k.price
from   shop_products k
left join shop_products_trans t
    on  t.products_id   = k.id
    and t.language_code = sys_context('APP_CTX', 'LANG');
```

## 8. Immutable table and temporal history

`/immutable` creates an append-only table by installing a trigger that raises an error on any UPDATE or DELETE. `/history` generates a shadow `_history` table and triggers that capture every row version with timestamp ranges.

**Input:**

```quicksql
audit_log /immutable
  entity    vc128 /nn
  entity_id num /nn
  operation vc10 /nn
  payload   clob

contract_versions /history
  contract_id num /nn /fk contracts
  content     clob /nn
  version_num num /nn
```

**Key generated DDL:**

```sql
create table audit_log (
    id          number generated by default on null as identity
                constraint audit_log_id_pk primary key,
    entity      varchar2(128 char) not null,
    entity_id   number not null,
    operation   varchar2(10 char) not null,
    payload     clob
);

-- Trigger blocks any UPDATE or DELETE
create or replace trigger trg_audit_log_insertonly
    before update or delete on audit_log
begin
    raise_application_error(-20055, 'audit_log is immutable');
end;
/

create table contract_versions (
    id              number generated by default on null as identity
                    constraint contract_versions_id_pk primary key,
    contract_id     number not null
                    constraint contract_versions_contract_id_fk
                    references contracts,
    content         clob not null,
    version_num     number not null,
    valid_from      timestamp,
    valid_to        timestamp
);

-- History shadow table captures every row state
create table contract_versions_history (
    id              number,
    contract_id     number,
    content         clob,
    version_num     number,
    valid_from      timestamp,
    valid_to        timestamp,
    operation       varchar2(10 char),
    changed_by      varchar2(255 char),
    changed_on      timestamp
);
```

## 9. SODA document collection

`/soda` creates a table with the fixed schema required by Oracle SODA: a RAW(16) document ID, timestamps, a version string, and a BLOB column with a `IS JSON` check.

**Input:**

```quicksql
sensor_readings /soda
events /soda /rest
```

**Key generated DDL:**

```sql
create table sensor_readings (
    id                varchar2(255 char)
                      constraint sensor_readings_id_pk primary key,
    created_on        timestamp,
    last_modified     timestamp,
    version           varchar2(255 char),
    json_document     blob
                      check (json_document is json)
);

create table events (
    id                varchar2(255 char)
                      constraint events_id_pk primary key,
    created_on        timestamp,
    last_modified     timestamp,
    version           varchar2(255 char),
    json_document     blob
                      check (json_document is json)
);

-- REST enablement for events
begin
    ords.enable_object(
        p_enabled      => true,
        p_schema       => sys_context('USERENV','CURRENT_SCHEMA'),
        p_object       => 'EVENTS',
        p_object_type  => 'TABLE',
        p_auto_rest_auth => false
    );
    commit;
end;
/
```

SODA tables are managed through the Oracle SODA API (Java, Python, REST) and are not meant to be queried relationally.

## 10. Flashback Data Archive

`/flashback` enables Flashback Data Archive on the table. An optional archive name can be specified; if omitted Oracle uses the default archive.

**Input:**

```quicksql
salaries /flashback payroll_fda
  employee_id /fk employees /nn
  amount      num(12,2) /nn
  effective   date /nn

config_values /flashback
  key   vc100 /nn /unique
  value clob
```

**Key generated DDL:**

```sql
create table salaries (
    id             number generated by default on null as identity
                   constraint salaries_id_pk primary key,
    employee_id    number not null
                   constraint salaries_employee_id_fk references employees,
    amount         number(12,2) not null,
    effective      date not null
);

alter table salaries flashback archive payroll_fda;

create table config_values (
    id      number generated by default on null as identity
            constraint config_values_id_pk primary key,
    key     varchar2(100 char) not null
            constraint config_values_key_unq unique,
    value   clob
);

alter table config_values flashback archive;
```

Flashback Data Archive requires an archive configured by a DBA (`CREATE FLASHBACK ARCHIVE`). It retains all historical row versions for queries like `AS OF TIMESTAMP`.

## 11. Sample data generation: /insert, /values, /constant

`/insert N` generates N random INSERT rows per table. `/values` constrains the random pool for a specific column. `/constant` pins a column to a fixed value. `inserts: false` suppresses all INSERTs globally (useful for production DDL scripts).

**Input:**

```quicksql
orders /insert 5
  customer_id num /nn /constant 42
  status      /values OPEN,PENDING,SHIPPED /nn
  region      vc20 /values EMEA,AMER,APAC
  total       num(10,2)

# inserts: true
```

**Key generated DDL (data portion):**

```sql
insert into orders (customer_id, status, region, total)
values (42, 'PENDING', 'APAC', 1847.23);

insert into orders (customer_id, status, region, total)
values (42, 'OPEN', 'EMEA', 392.10);

-- ... 3 more rows; customer_id always 42, status/region from the /values pools
```

To generate clean DDL without any INSERT statements:

```quicksql
orders /insert 100   -- directive is ignored when inserts is false
  name vc100

# inserts: false
```

## 12. Oracle SQL Annotations

Curly-brace `{...}` syntax adds Oracle SQL annotations to tables, columns, and views. The special `DESCRIPTION` key also generates a `COMMENT ON` statement. The `GROUP` key registers an annotation group.

**Input:**

```quicksql
employees {DESCRIPTION 'HR workforce table', Classification 'HR', GROUP 'PII'}
  name     vc200 /nn {DESCRIPTION 'Full legal name', Sensitivity 'Private'}
  email    vc200      {DESCRIPTION 'Work email address'}
  salary   num(12,2)  {Sensitivity 'Confidential'}
```

**Key generated DDL:**

```sql
create table employees (
    id        number generated by default on null as identity
              constraint employees_id_pk primary key,
    name      varchar2(200 char) not null
              annotations (DESCRIPTION 'Full legal name', Sensitivity 'Private'),
    email     varchar2(200 char)
              annotations (DESCRIPTION 'Work email address'),
    salary    number(12,2)
              annotations (Sensitivity 'Confidential')
)
annotations (DESCRIPTION 'HR workforce table', Classification 'HR', GROUP 'PII');

-- DESCRIPTION annotation also generates COMMENT ON
comment on table employees is 'HR workforce table';
comment on column employees.name  is 'Full legal name';
comment on column employees.email is 'Work email address';

-- GROUP annotation inserts into annotation group tables
insert into user_annotations_groups$ (group_name) values ('PII');
insert into user_annotations_group_members$ (group_name, object_name)
values ('PII', 'EMPLOYEES');
```

## 13. REST enablement with ORDS

`/rest` generates an `ords.enable_object` call to publish the table as a REST resource via Oracle REST Data Services.

**Input:**

```quicksql
products /rest /insert 3
  name  vc200 /nn
  price num(10,2) /nn
  sku   vc50 /unique

categories /rest
  label vc100 /nn /unique
```

**Key generated DDL:**

```sql
create table products (
    id      number generated by default on null as identity
            constraint products_id_pk primary key,
    name    varchar2(200 char) not null,
    price   number(10,2) not null,
    sku     varchar2(50 char)
            constraint products_sku_unq unique
);

create table categories (
    id      number generated by default on null as identity
            constraint categories_id_pk primary key,
    label   varchar2(100 char) not null
            constraint categories_label_unq unique
);

-- REST enablement block (one per /rest table)
begin
    ords.enable_object(
        p_enabled        => true,
        p_schema         => sys_context('USERENV','CURRENT_SCHEMA'),
        p_object         => 'PRODUCTS',
        p_object_type    => 'TABLE',
        p_auto_rest_auth => false
    );
    commit;
end;
/
```

ORDS must be installed and configured in the database. After running the script, the table is accessible at `https://<host>/ords/<schema>/products/`.

## 14. Table API (TAPI) ŌĆö simple and layered tiers

**Simple API** (`api: yes`): one `<table>_api` package with `get_row`, `insert_row`, `update_row`, `delete_row`.

**Layered API**: a stack of packages per table, selected by the tier argument on `/api`. No global setting needed ŌĆö the tier is self-contained.

### Simple API

```quicksql
products /api
  name  vc200 /nn
  price num(10,2)

# settings = { api: yes }
```

Generated: `products_api` ŌĆö single package with four procedures.

### Layered API ŌĆö all six tiers

```quicksql
-- tier: lookup
-- generated: items_apx
items /api lookup
  code  vc20 /nn
  label vc100 /nn

-- tier: lookup+hks
-- generated: categories_hks, categories_apx
categories /api lookup+hks
  code  vc20 /nn
  label vc100 /nn

-- tier: service
-- generated: orders_svc, orders_apx
-- _svc body embeds private DML (no _dal)
orders /api service
  customer_id num /nn
  total       num(12,2)

-- tier: service+hks
-- generated: invoices_hks, invoices_svc, invoices_apx
invoices /api service+hks
  order_id num /nn
  amount   num(12,2)

-- tier: full  (no hook stubs)
-- generated: employees_dal, employees_svc, employees_apx
employees /api full
  name  vc100 /nn
  email vc200 /nn

-- tier: full+hks  (default for bare /api)
-- generated: contracts_dal, contracts_hks, contracts_svc, contracts_apx
contracts /api full+hks
  title      vc200 /nn
  row_version num /nn
```

Package name anatomy for `contracts` with no prefix:

| Package | Role |
| --- | --- |
| `contracts_dal` | Data Access Layer ŌĆö raw DML (`get_by_id`, `insert_row`, `update_row`, `delete_row`) |
| `contracts_hks` | Hooks ŌĆö override stubs (`validate`, `before_insert`, `after_insert`, ŌĆ”) |
| `contracts_svc` | Service ŌĆö orchestrates DAL + hooks; public API (`create_rec`, `get`, `update_rec`, `delete_rec`) |
| `contracts_apx` | APEX interface ŌĆö thin scalar-parameter wrappers over `_svc` |

## 15. Audit log chain (`/auditlog`)

`/auditlog` on a table generates an `<table>_aud` package that captures every DML event
in an audit log table via `PRAGMA AUTONOMOUS_TRANSACTION`. The audit record persists even
if the outer transaction rolls back.

**Requirements:** the audit log table must exist in the schema (or be defined in the same
QSQL input), and both tables must use a layered tier.

```quicksql
-- 1. Define the audit log table (also gets layered API for querying)
app_audit_log /api full+hks
  entity     vc128 /nn
  entity_id  num   /nn
  operation  vc6   /nn
  old_values clob
  new_values clob

-- 2. Audited table ŌĆö /auditlog names the log table (without prefix)
employees /api full+hks /auditlog app_audit_log
  name        vc100 /nn
  email       vc200 /nn /unique
  row_version num   /nn
```

**Generated packages for `employees`:**

| Package | Role |
| --- | --- |
| `employees_dal` | Raw DML |
| `employees_hks` | Business rule stubs |
| `employees_svc` | Orchestration; calls `employees_aud` after each DML |
| `employees_apx` | APEX interface |
| `employees_aud` | Autonomous-transaction audit writer |

The `employees_aud` package:

- declares `g_enabled boolean := true` ŌĆö flip to `false` to suppress audit without code change
- contains a private `f_to_json` function that serialises the row to a JSON string
- `log_insert(p_row)`, `log_update(p_old_row, p_new_row)`, `log_delete(p_old_row)` delegate to a private `p_log` procedure that calls `app_audit_log_svc.create_rec` inside an autonomous transaction

> **Log table name:** `/auditlog` with no argument defaults to `app_audit_log`.
> Use the explicit form ŌĆö `/auditlog my_audit_log` ŌĆö when your log table has a
> different name.

## 16. Complete schema combining multiple features

A realistic schema that combines several features: prefix, audit columns, APEX, rowversion, FK constraints, check constraints, annotations, REST, and a duality view.

**Input:**

```quicksql
customers {DESCRIPTION 'Registered customers'}
  full_name  vc200 /nn {DESCRIPTION 'Legal full name'}
  email      vc200 /nn /unique /lower
  country    vc2
  tier       /check STANDARD,GOLD,PLATINUM /nn /default STANDARD

orders /rest /auditcols /rowversion
  customer_id /fk customers /nn
  status      /check OPEN,PROCESSING,SHIPPED,DELIVERED,CANCELLED /nn
  total       num(12,2) /nn
  notes       clob

order_items
  order_id   /fk orders /nn
  product_id /fk products /nn
  quantity   num /nn
  unit_price num(10,2) /nn

view customer_orders customers orders

# settings = {
#   prefix:    "shop_",
#   auditcols: yes,
#   apex:      yes,
#   rowversion: no,
#   drop:      no,
#   db:        "23c",
#   semantics: CHAR
# }
```

**Key generated DDL highlights:**

```sql
-- customers: annotation generates COMMENT ON
create table shop_customers (
    id          number generated by default on null as identity
                constraint shop_customers_id_pk primary key,
    full_name   varchar2(200 char) not null
                annotations (DESCRIPTION 'Legal full name'),
    email       varchar2(200 char) not null
                constraint shop_customers_email_unq unique,
    country     varchar2(2 char),
    tier        varchar2(4000 char) not null default 'STANDARD'
                constraint shop_customers_tier_ck
                check (tier in ('STANDARD','GOLD','PLATINUM')),
    created     date,
    created_by  varchar2(255 char),
    updated     date,
    updated_by  varchar2(255 char)
)
annotations (DESCRIPTION 'Registered customers');

comment on table shop_customers is 'Registered customers';
comment on column shop_customers.full_name is 'Legal full name';

-- orders: REST-enabled, audit columns, row_version

                    constraint shop_orders_id_pk primary key,
    customer_id     number not null
                    constraint shop_orders_customer_id_fk
                    references shop_customers,
    status          varchar2(4000 char) not null
                    constraint shop_orders_status_ck
                    check (status in ('OPEN','PROCESSING','SHIPPED','DELIVERED','CANCELLED')),

    updated_by      varchar2(255 char)

-- REST + audit + rowversion triggers generated automatically
-- Join view

       shop_customers.full_name  full_name,
```

## 17. Shared-schema multi-tenancy with `tenantid: yes`

The `tenantid: yes` setting implements the shared-schema (discriminator) multi-tenancy pattern. QuickSQL automatically wires the schema for tenant isolation, leaving only optional hardening steps for the developer.

**Supra-tenant vs tenant tables**

Tables marked `/notenantid` are shared across all tenants (lookup data, tenant master). All other tables receive tenant-scoped DDL.

**Input:**

```quicksql
subscription_plans /notenantid    -- supra-tenant lookup: shared, no TENANT_ID
  code  vc20 /nn /pk
  name  vc100 /nn
  price num(10,2) /nn

tenants /notenantid               -- master tenant table: supra-tenant
  name      vc200 /nn
  plan_code /fk subscription_plans /nn

customers                         -- tenant table: gets TENANT_ID NOT NULL automatically
  full_name vc200 /nn
  email     vc200 /nn /unique     -- scoped unique: becomes (tenant_id, email)
  country   vc3

orders                            -- tenant table
  customer_id /fk customers /nn  -- composite FK generated automatically
  status      /check OPEN,SHIPPED,CLOSED /nn /default OPEN
  total       num(14,2) /nn

order_lines                       -- tenant table
  order_id   /fk orders /nn /cascade
  sku        vc50 /nn
  qty        num /nn
  unit_price num(10,2) /nn

# settings = { prefix: "app_", tenantid: yes, auditcols: yes, db: "23c" }
```

**Key DDL generated:**

```sql
-- supra-tenant: no TENANT_ID, no tenant logic
create table app_subscription_plans (
    code   varchar2(20 char) not null constraint app_subscription_plans_code_pk primary key,
    name   varchar2(100 char) not null,
    price  number(10,2) not null
);

create table app_tenants (
    id    number generated by default on null as identity
          constraint app_tenants_id_pk primary key,
    name  varchar2(200 char) not null,
    ...
);

-- tenant table: TENANT_ID NOT NULL injected as second column
create table app_customers (
    id          number ... constraint app_customers_id_pk primary key,
    tenant_id   number not null,           -- ŌåÉ added by tenantid: yes, NOT NULL by default
    full_name   varchar2(200 char) not null,
    email       varchar2(200 char) not null,
    country     varchar2(3 char),
    ...
);

-- email unique scoped to tenant (not global)
create unique index app_customers_tid_email_uix
    on app_customers (tenant_id, email);

-- FK index: tenant_id as leading column
create index app_customers_i1 on app_customers (tenant_id, customer_id);

-- auto-FK: tenant_id ŌåÆ tenants master table (generated because tenants is in the same script)
alter table app_customers add constraint app_customers_tenant_id_fk
    foreign key (tenant_id) references app_tenants (id);

-- (tenant_id, id) unique on customers: on-demand, prerequisite for the composite FK from orders
create unique index app_customers_tid_id_uix
    on app_customers (tenant_id, id);
alter table app_customers add constraint app_customers_tid_id_uq
    unique (tenant_id, id) using index app_customers_tid_id_uix;

-- composite FK from orders to customers: prevents cross-tenant references at DB level
alter table app_orders add constraint app_orders_customer_id_fk
    foreign key (tenant_id, customer_id)
    references app_customers (tenant_id, id);

-- composite FK with ON DELETE CASCADE (from /cascade on order_lines.order_id)
alter table app_order_lines add constraint app_order_lines_order_id_fk
    foreign key (tenant_id, order_id)
    references app_orders (tenant_id, id) on delete cascade;
```

**What QuickSQL generates automatically:**

| Feature | How to declare |
|---|---|
| `TENANT_ID NUMBER NOT NULL` on all tenant tables | `tenantid: yes` in settings |
| Nullable `TENANT_ID` (for sample data) | Add `/insert N` to the table |
| Auto-FK `tenant_id ŌåÆ tenants(id)` | Have a `tenants` table in the same script |
| Auto-FK to a custom-named master table | `tenantref: "workspaces"` in settings |
| Composite FK `(tenant_id, fk_col) ŌåÆ target(tenant_id, id)` | Any `/fk` between tenant tables |
| `ON DELETE CASCADE` on composite FK | Add `/cascade` to the FK column |
| `(tenant_id, id)` unique on target table | Generated on-demand with FK above |
| `(tenant_id, col)` unique from `/unique` | Add `/unique` to the column |
| `(tenant_id, col)` index from `/idx` | Add `/idx` to the column |
| Supra-tenant table (no TENANT_ID) | Add `/notenantid` to the table |
| Simple FK to supra-tenant table | Automatic when target is `/notenantid` |

**Remaining manual steps** (only when needed):

```sql
-- Only for tables with /insert N (tenant_id is nullable there):
update app_orders set tenant_id = 1 where tenant_id is null;
commit;
alter table app_orders modify tenant_id not null;

-- When the tenants table is in a separate QSQL script (auto-FK not generated):
alter table app_customers add constraint app_customers_tenant_id_fk
    foreign key (tenant_id) references app_tenants (id);
```

See [Multi-Tenant Design](./multitenant-design.md) for the complete enterprise guide: VPD, filtered views, application context, partitioning, and the hardening checklist.

## 18. Schema migration with `toDiff`

**Interactive (editor):** write both schema versions in the same editor separated by `# ---`. The DDL panel switches to migration mode automatically ŌĆö no extra tabs or buttons needed. Use **Ōćä Compare versions** to insert the delimiter.

```quicksql
employees
  name vc100 /nn
  email vc200

# ---

employees
  name vc100 /nn
  email vc200 /nn
  phone vc50

departments
  dept_name vc100 /nn
```

The panel shows the migration script, a status bar with counts, and warning bars for destructive operations. See the [Schema Migration](quick-sql-grammar.md#schema-migration) section of the grammar reference for all details.

**Programmatic:** `toDiff(oldQsql, newQsql)` computes the incremental Oracle DDL migration script between two schema versions. The output is a `DiffResult` containing a runnable SQL script, per-statement metadata, and warnings for destructive or lossy operations.

### Adding a column and a new table

**v1 (old):**

```quicksql
employees
  name vc100 /nn
  email vc200
```

**v2 (new):**

```quicksql
employees
  name  vc100 /nn
  email vc200 /nn
  phone vc50

departments
  dept_name vc100 /nn
```

**Call:**

```javascript
const result = toDiff(v1, v2);
```

**Generated `result.sql` (key excerpts):**

```sql
-- ============================================================
-- QuickSQL Migration Script
-- Generated : 2026-05-06T10:00:00.000Z
-- ============================================================
-- ŌÜĀ MANUAL STEPS REQUIRED (statementsRequiringIntervention = 1)
--   [employees.email]
--   Ensure all rows have a non-null value before executing.
-- ============================================================

create table departments (
    id           raw(16)        default sys_guid() not null,
    dept_name    varchar2(100 char) not null,
    --
    constraint departments_pk primary key (id)
);

alter table employees add (phone varchar2(50 char));

-- ŌÜĀ MANUAL INTERVENTION REQUIRED
-- Ensure all rows have a non-null value before executing.
-- alter table employees modify (email not null);
```

**`result.warnings`:**

```json
[
  { "level": "DESTRUCTIVE", "table": "employees", "column": "email",
    "message": "adding NOT NULL on email ŌĆö requires manual verification",
    "requiresManualIntervention": true }
]
```

**`result.summary`:**

```json
{ "tablesAdded": 1, "tablesDropped": 0, "tablesModified": 1,
  "statementsTotal": 4, "statementsRequiringIntervention": 1, "warningsTotal": 1 }
```

### Dropping a column

```quicksql
-- v1
employees
  name  vc100 /nn
  email vc200
  phone vc50

-- v2 (phone removed)
employees
  name  vc100 /nn
  email vc200
```

Generated migration (two-step Oracle pattern):

```sql
alter table employees set unused column phone;
-- [MAINTENANCE] safe to defer to a maintenance window
alter table employees drop unused columns;
```

Warning emitted: `DESTRUCTIVE ŌĆö column dropped: phone`.

### Rename detection

When exactly one column is dropped and one of the same base type is added, `toDiff` emits a `rename_hint` instead of a destructive DROP + ADD:

```quicksql
-- v1: old_email  vc200
-- v2: new_email  vc200  (same base type ŌåÆ suspected rename)
```

```sql
-- alter table employees rename column old_email to new_email;
```

Warning: `INFO ŌĆö suspected rename: old_email ŌåÆ new_email (verify before applying)`. No destructive DDL is generated; apply the commented statement manually after verifying.

### 23c syntax

Embed `# settings = {"db": "23c"}` in the QSQL strings to get `IF EXISTS` / `IF NOT EXISTS` syntax and skip the PL/SQL idempotency wrappers:

```javascript
const db23 = '\n# settings = {"db": "23c"}';
const result = toDiff(v1 + db23, v2 + db23);
// drop table  if exists employees cascade constraints;
// create index if not exists employees_email_i on employees (email);
```

## 19. Layered TAPI ŌĆö per-table tier selection

Different tables in the same schema often need different levels of API complexity.
The `/api <tier>` syntax selects the tier per table independently of the global default.

```quicksql
-- reference data: thin APEX interface only
-- generated: lookup_codes_apx
lookup_codes /api lookup
  code  vc20 /nn /unique
  label vc100 /nn

-- business entity with hook stubs but no DAL
-- generated: orders_hks, orders_svc, orders_apx
orders /api service+hks
  customer_id num /nn
  total       num(12,2)

-- core entity: full stack without hooks
-- generated: products_dal, products_svc, products_apx
products /api full
  name  vc100 /nn
  price num(12,2)

-- top-level entity: full stack with hooks
-- generated: employees_dal, employees_hks, employees_svc, employees_apx
employees /api full+hks
  name        vc100 /nn
  email       vc200 /nn /unique
  row_version num   /nn

# settings = { api: layered }
```

Numeric shorthand: `1`=`lookup`, `1h`=`lookup+hks`, `2`=`service`, `2h`=`service+hks`,
`3`=`full`, `3h`=`full+hks`. Both forms are equivalent: `/api 3h` and `/api full+hks`.

## 20. Layered TAPI ŌĆö degradation (absorbed layers)

When a lower layer is absent, its logic is absorbed as private procedures in the package above it. This lets you start with a leaner tier and upgrade later without changing calling code.

**Service tier (no `_dal` ŌĆö DML absorbed into `_svc`):**

```quicksql
-- _svc body contains private p_get_by_id, p_insert_row, etc.
departments /api service
  name    vc100 /nn
  budget  num(15,2)
```

The generated `departments_svc` body begins with:

```sql
-- private DML (absorbed from absent _dal)

function p_get_by_id (p_id in departments.id%type) return departments%rowtype is ...
procedure p_insert_row (p_row in out nocopy departments%rowtype) is ...
```

**Service+hks tier ŌĆö `_hks` id type when no `_dal`:**

When `_dal` is absent, the hooks package uses the table anchor type directly:

```sql
-- service+hks: no _dal present
procedure before_delete (p_id in departments.id%type);

-- full+hks: _dal present ŌåÆ use its t_id subtype
procedure before_delete (p_id in departments_dal.t_id);
```

## 21. Layered TAPI ŌĆö REST interface (`ifc: rest`)

Use `ifc: rest` to generate ORDS handler packages (`_rst`) instead of APEX interface packages (`_apx`). Use `ifc: both` to generate both.

**Input:**

```quicksql
employees /api full+hks
  name        vc100 /nn
  email       vc200 /nn /unique
  row_version num /nn

# settings = { api: layered, ifc: rest }
```

The generated `employees_rst` package:

- **No-parameter procedures**: `get`, `ins`, `upd`, `del` ŌĆö all parameters are ORDS bind variables.
- **Input**: `:body_text` (JSON body for ins/upd), `:p_id` (path parameter for get/upd/del).
- **Output**: `:status` (HTTP status code), `htp.p(json_object(...))` for response body.

```sql
create or replace package employees_rst as
    procedure get;
    procedure ins;
    procedure upd;
    procedure del;
end employees_rst;
/
```

For dual access (APEX + REST simultaneously):

```quicksql
# settings = { ifc: both }

employees /api full+hks
  name vc100 /nn
```

This generates both `employees_apx` and `employees_rst`.

## 22. IBM Db2 ŌĆö basic DDL

**Key generated DDL:**

```sql
--#SET TERMINATOR @

create table departments (
    id          integer not null
                    generated always as identity (start with 1 increment by 1)
```

## 23. IBM Db2 ŌĆö triggers and audit columns

The `/auditcols` directive generates Db2-compatible BEFORE INSERT and BEFORE UPDATE triggers using SQL PL syntax.

**Input:**

```quicksql
employees /auditcols
  name    vc100 /nn
  salary  num(12,2)

# settings = {"dialect":"db2","inserts":"no"}
```

**Key generated DDL:**

```sql
--#SET TERMINATOR @

create table employees (
    id          integer not null
                    generated always as identity (start with 1 increment by 1)
                    constraint employees_id_pk primary key,
    name        varchar(100) not null,
    salary      decimal(12,2),
    created_by  varchar(255),
    created     timestamp,
    updated_by  varchar(255),
    updated     timestamp
) @

create or replace trigger employees_biu
before insert on employees
referencing new as n
for each row
begin
    if inserting then
        set n.created_by = current user;
        set n.created    = current timestamp;
    end if;
    set n.updated_by = current user;
    set n.updated    = current timestamp;
end @

create or replace trigger employees_buu
before update on employees
referencing new as n old as o
for each row
begin
    set n.updated_by = current user;
    set n.updated    = current timestamp;
end @
```

Db2 SQL PL trigger conventions:
- `REFERENCING NEW AS n` (insert), `REFERENCING NEW AS n OLD AS o` (update) ŌĆö no `:new.` / `:old.` Oracle syntax.
- `SET n.col = value` ŌĆö no `:=` assignment operator.
- `CURRENT TIMESTAMP` and `CURRENT USER` ŌĆö no `SYSDATE` or `v('APP_USER')`.
- Each object ends with `@` (the custom terminator set at the top).

## 24. IBM Db2 ŌĆö layered TAPI with schema-based procedures

The `/api` directive generates a layered Table API in Db2 SQL PL.  Because Db2 has no packages, each tier becomes a dedicated **schema** containing `CREATE OR REPLACE PROCEDURE` statements.

**Input:**

```quicksql
employees /api full+hks
  name    vc100 /nn
  email   vc200 /nn /unique

# settings = {"dialect":"db2","inserts":"no"}
```

**Generated schema layout (`full+hks`):**

| Schema | Contents |
|--------|----------|
| `employees_dal` | `p_get_by_id`, `p_insert_row`, `p_update_row`, `p_delete_row` |
| `employees_hks` | `p_validate`, `p_before_insert`, `p_after_insert`, `p_before_update`, `p_after_update`, `p_before_delete`, `p_after_delete` |
| `employees_svc` | `get`, `ins`, `upd`, `del` ŌĆö orchestration layer, calls DAL + HKS |
| `employees_rst` | `get`, `ins`, `upd`, `del` ŌĆö REST interface, returns HTTP status codes |

**Selected generated DDL:**

```sql
create schema employees_dal @
create schema employees_hks @
create schema employees_svc @
create schema employees_rst @

-- DAL: return a cursor (DYNAMIC RESULT SETS)
create or replace procedure employees_dal.p_get_by_id (
    in  p_id     integer,
    out p_status varchar(10)
)
dynamic result sets 1
language sql
begin
    declare c cursor with return for
        select * from employees where id = p_id;
    open c;
    set p_status = 'OK';
end @

-- DAL: insert and return the new PK
create or replace procedure employees_dal.p_insert_row (
    in  p_name    varchar(100),
    in  p_email   varchar(200),
    out p_id      integer,
    out p_status  varchar(10)
)
language sql
begin
    insert into employees (name, email)
    values (p_name, p_email);
    set p_id = identity_val_local();
    set p_status = 'OK';
end @

-- HKS: validation stub (fill in business rules)
create or replace procedure employees_hks.p_validate (
    in p_action  varchar(10),
    in p_context varchar(255)
)
language sql
begin
    -- add validation logic here
end @

-- SVC: orchestrates HKS validate + DAL insert
create or replace procedure employees_svc.ins (
    in  p_name    varchar(100),
    in  p_email   varchar(200),
    out p_id      integer,
    out p_status  varchar(10)
)
language sql
begin
    declare exit handler for sqlexception
    begin
        set p_status = 'ERROR';
    end;
    call employees_hks.p_validate('INSERT', '');
    call employees_dal.p_insert_row(p_name, p_email, p_id, p_status);
end @

-- RST: REST interface ŌĆö returns HTTP status codes
create or replace procedure employees_rst.ins (
    in  p_name    varchar(100),
    in  p_email   varchar(200),
    out p_id      integer,
    out p_status  integer,
    out p_result  varchar(4000)
)
language sql
begin
    declare v_svc_status varchar(10);
    declare exit handler for sqlexception
    begin
        get diagnostics exception 1 p_result = message_text;
        set p_status = 500;
        set p_result = json_object('error': p_result);
    end;
    call employees_svc.ins(p_name, p_email, p_id, v_svc_status);
    set p_status = 201;
    set p_result = json_object('id': p_id);
end @
```

**Available tier values** (same names as Oracle):

| Tier | Schemas generated |
|------|------------------|
| `full+hks` (default) | `_dal` + `_hks` + `_svc` + `_rst` |
| `full` | `_dal` + `_svc` + `_rst` |
| `service+hks` | `_hks` + `_svc` + `_rst` (SVC absorbs DML privately) |
| `service` | `_svc` + `_rst` (SVC owns DML directly) |
| `lookup+hks` | `_hks` + `_rst` (read-only, hooks, RST absorbs get) |
| `lookup` | `_rst` only (read-only REST) |

Numeric aliases also work: `3h` = `full+hks`, `3` = `full`, `2` = `service`, `1` = `lookup`.
